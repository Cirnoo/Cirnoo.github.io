### 快排

```cpp
class Algorithm
{
public:
	void qsort(vector<int> & vec, int l, int h)
	{
		if (l >= h)
			return;
		int mid = partition(vec, l, h);
		qsort(vec, l, mid - 1);
		qsort(vec, mid + 1, h);
	}

private:
	int partition(vector<int> & arr, int l, int h)
	{
		int key = arr[l];
		while (l < h)
		{
			while (l < h && arr[h] >= key)
				--h;
			arr[l] = arr[h];
			while (l < h && arr[l] <= key)
				++l;
			arr[h] = arr[l];
		}
		arr[l] = key;
		return l;
	}
};
```
